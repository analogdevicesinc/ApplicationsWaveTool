95  std::vector<uint8_t> pack() {
96  std::vector<uint8_t> out_pkt;
97 
98  for (unsigned int i = 0; i < offsetof(m2m2_hdr_t, data); i++) {
99  out_pkt.push_back(((uint8_t *)(&this->header))[i]);
100  }
101 
102  for (unsigned int i = 0; i < sizeof(this->payload); i++) {
103  out_pkt.push_back(((uint8_t *)(&this->payload))[i]);
104  }

111  std::vector<uint8_t> pack_ping() {
112  std::vector<uint8_t> out_pkt;
113 
114  for (unsigned int i = 0; i < offsetof(m2m2_hdr_t, data); i++) {
115  out_pkt.push_back(((uint8_t *)(&this->header))[i]);
116  }
117 
118  for (unsigned int i = 0; i < sizeof(this->payload); i++) {
119  out_pkt.push_back(((uint8_t *)(&this->payload))[i]);
120  }
121  for (unsigned int i = 0; i <= (this->header.length - sizeof(this->header) - sizeof(this->payload)); i++) {
122  out_pkt.push_back(0);
123  }
124 
125  return out_pkt;
126  }

131  void unpack(std::vector<uint8_t> &bytes) {
132 
133  //return if the packet is empty
134  if (bytes == std::vector<uint8_t>{0})
135  return;
136  // Copy the header into our internal structure objects
137  this->header.src = static_cast<M2M2_ADDR_ENUM_t> (BYTE_SWAP_16(this->header.src));
138  this->header.dest = static_cast<M2M2_ADDR_ENUM_t> (BYTE_SWAP_16(this->header.dest));
139  this->header.length = static_cast<M2M2_ADDR_ENUM_t> (BYTE_SWAP_16(this->header.length));
140  this->header.checksum = static_cast<M2M2_ADDR_ENUM_t> (BYTE_SWAP_16(this->header.checksum));
141  std::memcpy(&this->payload, &bytes[offsetof(m2m2_hdr_t, data)], sizeof(T));
142  }
